# Основной сайт с HTTPS (лендинг, без WordPress)
server {
    listen 443 ssl http2;
    server_name bestaibots.ru www.bestaibots.ru;

    root /var/www/bestaibots;
    index index.html index.htm;
    ssl_certificate /etc/letsencrypt/live/bestaibots.ru-0001/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/bestaibots.ru-0001/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    # Защита от атак и оптимизация
    add_header X-Frame-Options SAMEORIGIN;
    add_header X-Content-Type-Options nosniff;
    add_header X-XSS-Protection "1; mode=block";
    client_max_body_size 20M;
    client_body_timeout 12;
    send_timeout 10;

    # Включение сжатия Gzip
    gzip on;
    gzip_disable "msie6";
    gzip_vary on;
    gzip_proxied any;
    gzip_comp_level 6;
    gzip_types text/plain text/css application/json application/javascript text/xml application/xml application/xml+rss text/javascript;

    # Основной блок для HTML-лендинга
    location / {
        try_files $uri $uri/ =404;
    }

    # Запрет на просмотр скрытых файлов
    location ~ /\.ht {
        deny all;
    }

    location /sub/ {
        alias /var/www/subscriptions/;
        autoindex on;
        add_header Content-Type text/plain;
        add_header Access-Control-Allow-Origin *;
    }


}
